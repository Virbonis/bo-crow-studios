import { mapDispatchToProps } from './path/to/your/component';
import actions from 'redux/shareholder-buyback/actions';

describe('mapDispatchToProps', () => {
  let dispatch;
  let mappedProps;

  beforeEach(() => {
    dispatch = jest.fn();
    mappedProps = mapDispatchToProps(dispatch);
  });

  it('should dispatch the CREATE action with the correct payload', () => {
    const payload = { /* Your payload here */ };
    const successCallback = jest.fn(); // Mock success callback function

    // Call the Create function
    mappedProps.Create(payload, successCallback);

    // Assert that the dispatch was called with the correct action
    expect(dispatch).toHaveBeenCalledWith({
      type: actions.CREATE,
      payload,
      successCallback,
      source: '',
    });
  });
});
